4/// Core AES decryption function. Decrypts 16 byte block using provided round keys.
5#[inline(always)]
6pub fn decrypt_block(ciphertext: &[u8; 16], round_keys: &[[u8; 16]]) -> [u8; 16] {
7    let mut state = *ciphertext;
8    let num_rounds = round_keys.len();
9
10    add_round_key(&mut state, &round_keys.last().unwrap());
11
12    for round_key in round_keys[1..num_rounds - 1].iter().rev() {
13        shift_rows_inv(&mut state);
14        sub_bytes_inv(&mut state);
15        add_round_key(&mut state, round_key);
16        mix_columns_inv(&mut state);
17    }
18
19    shift_rows_inv(&mut state);
20    sub_bytes_inv(&mut state);
21    add_round_key(&mut state, &round_keys[0]);
22
23    state
24}

26/// Inverse SubBytes step. Each byte is substituted using the inverse SBOX.
27#[inline(always)]
28pub(crate) fn sub_bytes_inv(state: &mut [u8; 16]) {
29    for byte in state {
30        *byte = SBOX_INV[*byte as usize];
31    }
32}

34/// Inverse ShiftRows step. 
35/// The 0th row does not change. 
36/// The first row shifts right by one position. 
37/// The second row shifts right by two positions.
38/// The third row shifts right by three positions.
39#[inline(always)]
40fn shift_rows_inv(state: &mut [u8; 16]) {
41    // trivial implementation to improve efficiency
42
43    let s = *state;
44
45    // row 0 (indices 0,4,8,12): unchanged
46    state[0]  = s[0];
47    state[4]  = s[4];
48    state[8]  = s[8];
49    state[12] = s[12];
50
51    // row 1 (1,5,9,13): right rotate by 1
52    state[1]  = s[13];
53    state[5]  = s[1];
54    state[9]  = s[5];
55    state[13] = s[9];
56
57    // row 2 (2,6,10,14): right rotate by 2
58    state[2]  = s[10];
59    state[6]  = s[14];
60    state[10] = s[2];
61    state[14] = s[6];
62
63    // row 3 (3,7,11,15): right rotate by 3
64    state[3]  = s[7];
65    state[7]  = s[11];
66    state[11] = s[15];
67    state[15] = s[3];
68}

70/// Inverse MixColumns step. Each column is multiplied by a constant matrix using Galois field multiplication.
71/// [ b0 ]      [ 14  11  13  09 ]  [ d0 ]
72/// | b1 |  =   | 09  14  11  13 |  | d1 |
73/// | b2 |      | 13  09  14  11 |  | d2 |
74/// [ b3 ]      [ 11  13  09  14 ]  [ d3 ]
75#[inline(always)]
76fn mix_columns_inv(state: &mut [u8; 16]) {
77    // optimisation technique from https://crypto.stackexchange.com/a/71206
78    for col in 0..4 {
79        let i = col * 4;
80        let (a, b, c, d) = (state[i], state[i + 1], state[i + 2], state[i + 3]);
81        let x = dbl(a ^ b ^ c ^ d); /* 2a + 2b + 2c + 2d */
82        let y = dbl(x ^ a ^ c); /* 6a + 4b + 6c + 4d */
83        let z = dbl(x ^ b ^ d); /* 4a + 6b + 4c + 6d */
84        state[i + 0] = dbl(y ^ a ^ b) ^ b ^ c ^ d; /* 14a + 11b + 13c + 09d */
85        state[i + 1] = dbl(z ^ b ^ c) ^ c ^ d ^ a; /* 09a + 14b + 11c + 13d */
86        state[i + 2] = dbl(y ^ c ^ d) ^ d ^ a ^ b; /* 13a + 09b + 14c + 11d */
87        state[i + 3] = dbl(z ^ d ^ a) ^ a ^ b ^ c; /* 11a + 13b + 09c + 14d */
88    }
89}
