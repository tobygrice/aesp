7/// Core counter encryption and decryption implementation.
8pub fn ctr_core(
9    input: &[u8],
10    round_keys: &[[u8; 16]],
11    iv: &[u8; 12],
12    ctr_start: u32,
13) -> Result<Vec<u8>> {
14    if input.is_empty() {
15        return Ok(Vec::new());
16    }
17
18    // check if counter will overflow
19    let num_blocks = u32::try_from((input.len() + 15) / 16).map_err(|_| Error::CounterOverflow)?;
20    ctr_start
21        .checked_add(num_blocks - 1)
22        .ok_or(Error::CounterOverflow)?;
23
24    let mut output = vec![0u8; input.len()];
25    if input.len() > crate::aesp::modes::util::PARALLEL_THRESHOLD {
26        // encrypt in parallel if size exceeds threshold
27        output
28            .par_chunks_mut(16)
29            .zip(input.par_chunks(16))
30            .enumerate()
31            .for_each(|(i, (out_chunk, in_chunk))| {
32                let ctr = ctr_start + i as u32; // overflow already checked above
33                let block = ctr_block(iv, ctr); // form block from iv + ctr
34                // xor each element of input chunk (1-16 bytes) with encrypted ctr block
35                let keystream = encrypt_block(&block, round_keys);
36
37                // XOR bytes of block (last chunk may be lt 16)
38                for j in 0..in_chunk.len() {
39                    out_chunk[j] = keystream[j] ^ in_chunk[j];
40                }
41            });
42    } else {
43        // input len below threshold, encrypt serially
44        output
45            .chunks_mut(16)
46            .zip(input.chunks(16))
47            .enumerate()
48            .for_each(|(i, (out_chunk, in_chunk))| {
49                let ctr = ctr_start + i as u32; // overflow already checked above
50                let block = ctr_block(iv, ctr); // form block from iv + ctr
51                // xor each element of input chunk (1-16 bytes) with encrypted ctr block
52                let keystream = encrypt_block(&block, round_keys);
53
54                // XOR bytes of block (last chunk may be lt 16)
55                for j in 0..in_chunk.len() {
56                    out_chunk[j] = keystream[j] ^ in_chunk[j];
57                }
58            });
59    }
60
61    Ok(output)
62}

65mod test_ctr {
66    use super::*;
67    use crate::aesp::modes::util::test_util::{
68        CTR_IV, CTR_START, KEY_128, KEY_192, KEY_256, PLAINTEXT, hex_to_bytes,

70    use crate::{Cipher, Key};
71
72    #[test]
73    fn aes_ctr_128_encrypt() -> Result<()> {
74        let expected = hex_to_bytes(
75            "
76        874d6191b620e3261bef6864990db6ce\
77        9806f66b7970fdff8617187bb9fffdff\
78        5ae4df3edbd5d35e5b4f09020db03eab\
79        1e031dda2fbe03d1792170a0f3009cee",
80        );
81
82        let key = Key::try_from_slice(&KEY_128)?;
83        let cipher = Cipher::new(&key);
84        let encrypted = ctr_core(&PLAINTEXT, cipher.round_keys(), &CTR_IV, CTR_START)?;
85
86        assert_eq!(
87            expected, encrypted,
88            "encypted result does not match expected"
89        );
90        Ok(())
91    }
