4#[inline(always)]
5pub(crate) fn ctr_block(iv: &[u8; 12], ctr: u32) -> [u8; 16] {
6    let cb = ctr.to_be_bytes();
7    [
8        iv[00], iv[01], iv[02], iv[03], //
9        iv[04], iv[05], iv[06], iv[07], //
10        iv[08], iv[09], iv[10], iv[11], //
11        cb[00], cb[01], cb[02], cb[03], //
12    ]
13}

15#[inline(always)]
16pub(crate) fn mul_x(v: u128) -> u128 {
17    let lsb = v & 1;
18    let mut v2 = v >> 1;
19    v2 ^= GHASH_R & (0u128.wrapping_sub(lsb));
20    v2
21}

23#[inline(always)]
24pub(crate) fn mul_x4(mut v: u128) -> u128 {
25    v = mul_x(v);
26    v = mul_x(v);
27    v = mul_x(v);
28    v = mul_x(v);
29    v
30}

34    pub fn hex_to_bytes(s: &str) -> Vec<u8> {
35        let s = s.trim();
36        assert!(s.len() % 2 == 0, "hex string must have even length");
37        (0..s.len())
38            .step_by(2)
39            .map(|i| u8::from_str_radix(&s[i..i + 2], 16).unwrap())
40            .collect()
41    }
