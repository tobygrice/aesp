8/// Provides encryption and decryption functions for AES in modes [ECB](crate::Cipher::encrypt_ecb), [CTR](crate::Cipher::encrypt_ctr), and [GCM](crate::Cipher::encrypt_gcm).
9/// Instantiated with an AES [Key], which is expanded into round keys and stored in the instance.
10/// 
11/// ## Examples
12/// ```
13/// # fn main() -> aesp::Result<()> {
14/// use aesp::{Key, Cipher};
15/// 
16/// // Instantiate random key:
17/// let rk_256 = Key::rand_key_256()?;
18/// 
19/// // Instantiate AESP cipher using the key:
20/// let cipher = Cipher::new(&rk_256);
21/// # Ok(())
22/// # }
23/// ```
24#[derive(Clone, Eq, PartialEq, Debug)]
25pub struct Cipher {
26    round_keys: Vec<[u8; 16]>,
27}
28
29impl Cipher {
30    /// Generates round keys from provided key and stores in the returned instance.
31    pub fn new(key: &Key) -> Self {
32        Self {
33            round_keys: Self::expand_key(key),
34        }
35    }

37    /// Getter for internal round keys. Returned as a slice of 16-byte arrays.
38    pub fn round_keys(&self) -> &[[u8; 16]] {
39        &self.round_keys
40    }

42    /// **Electronic codebook** encryption.
43    ///
44    /// Encrypts each 16-byte block entirely independently and chains them together. 
45    /// Pads input to a multiple of 16 bytes using PKCS#7 padding.
46    /// **Vulnerable to pattern emergence in the ciphertext.**
47    /// 
48    /// ## Examples
49    /// ```
50    /// # fn main() -> aesp::Result<()> {
51    /// # use aesp::{Key, Cipher};
52    /// # let rk_256 = Key::rand_key_256()?;
53    /// # let cipher = Cipher::new(&rk_256);
54    /// let plaintext = ("Hello, World!").as_bytes();
55    /// let ciphertext = cipher.encrypt_ecb(&plaintext);
56    /// # Ok(())
57    /// # }
58    /// ```
59    pub fn encrypt_ecb(&self, plaintext: &[u8]) -> Vec<u8> {
60        ecb_core_enc(&pad(plaintext), &self.round_keys).unwrap() // safe unwrap, input is always padded
61    }

63    /// **Electronic codebook** decryption.
64    /// 
65    /// Assumes plaintext was PKCS#7 padded before encryption and unpads automatically.
66    /// Throws error if last block does not match PKCS#7 format or input is not a multiple of 16 bytes.
67    /// 
68    /// ## Examples
69    /// ```
70    /// # fn main() -> aesp::Result<()> {
71    /// # use aesp::{Key, Cipher};
72    /// # let rk_256 = Key::rand_key_256()?;
73    /// # let cipher = Cipher::new(&rk_256);
74    /// let plaintext = ("Hello, World!").as_bytes();
75    /// let ciphertext = cipher.encrypt_ecb(&plaintext);
76    /// let decrypted = cipher.decrypt_ecb(&ciphertext)?;
77    /// assert_eq!(decrypted, plaintext);
78    /// # Ok(())
79    /// # }
80    /// ```
81    pub fn decrypt_ecb(&self, ciphertext: &[u8]) -> Result<Vec<u8>> {
82        let mut ct = ecb_core_dec(ciphertext, &self.round_keys)?;
83        unpad(&mut ct)?;
84        Ok(ct)
85    }

87    /// **Counter mode** encryption.
88    ///
89    /// Generates a random 12-byte initialisation vector (IV).
90    /// For each 16-byte block of plaintext:
91    /// 1. 4-byte counter is incremented (starts at zero).
92    /// 2. Counter is appended to 12-byte IV to form a 16-byte block.
93    /// 3. The `IV || Counter` block is encrypted using the round keys.
94    /// 4. The plaintext block is `XOR`'d with the encrypted counter block.
95    ///
96    /// **Important**: the same IV must never be reused with the same key. 96 bits is
97    /// sufficiently large to assume uniqueness when randomly generated.
98    ///
99    /// Output is formatted as `IV (12 bytes) || Ciphertext`
100    /// 
101    /// ## Examples
102    /// ```
103    /// # fn main() -> aesp::Result<()> {
104    /// # use aesp::{Key, Cipher};
105    /// # let rk_256 = Key::rand_key_256()?;
106    /// # let cipher = Cipher::new(&rk_256);
107    /// let plaintext = ("Hello, World!").as_bytes();
108    /// let ciphertext = cipher.encrypt_ctr(&plaintext)?;
109    /// # Ok(())
110    /// # }
111    /// ```
112    pub fn encrypt_ctr(&self, plaintext: &[u8]) -> Result<Vec<u8>> {
113        // generate IV and prepend to ciphertext
114        let iv = random_iv()?;
115        let mut ciphertext = Vec::with_capacity(12 + plaintext.len());
116        ciphertext.extend_from_slice(&iv);
117        ciphertext.append(&mut ctr_core(plaintext, &self.round_keys, &iv, 0)?);
118        Ok(ciphertext)
119    }
120
121    /// **Counter mode** decryption.
122    ///
123    /// Assumes format matches output of encryption: `IV (12 bytes) || Ciphertext`
124    /// 
125    /// ## Examples
126    /// ```
127    /// # fn main() -> aesp::Result<()> {
128    /// # use aesp::{Key, Cipher};
129    /// # let rk_256 = Key::rand_key_256()?;
130    /// # let cipher = Cipher::new(&rk_256);
131    /// let plaintext = ("Hello, World!").as_bytes();
132    /// let ciphertext = cipher.encrypt_ctr(&plaintext)?;
133    /// let decrypted = cipher.decrypt_ctr(&ciphertext)?;
134    /// assert_eq!(decrypted, plaintext);
135    /// # Ok(())
136    /// # }
137    /// ```
138    pub fn decrypt_ctr(&self, ciphertext: &[u8]) -> Result<Vec<u8>> {
139        // extract and remove IV from ciphertext
140        if ciphertext.len() < 12 {
141            return Err(Error::InvalidCiphertext {
142                len: ciphertext.len(),
143                context: "CTR: missing 12-byte IV",
144            });
145        }
146
147        let (iv_bytes, ciphertext) = ciphertext.split_at(12);
148        let mut iv = [0u8; 12];
149        iv.copy_from_slice(iv_bytes);
150
151        ctr_core(ciphertext, &self.round_keys, &iv, 0)
152    }

154    /// **Galois/counter mode** encryption.
155    ///
156    /// Encrypts using counter mode and generates a cryptographic tag to verify the
157    /// message has not been modified.
158    ///
159    /// Also accepts optional additional authenticated data (AAD), which is included in the computation of the
160    /// tag but **not encrypted**.
161    ///
162    /// Output is formatted as `IV (12 bytes) || AAD length (4 bytes) || AAD || Ciphertext || Tag (16 bytes)`
163    /// 
164    /// ## Examples
165    /// ```
166    /// # fn main() -> aesp::Result<()> {
167    /// # use aesp::{Key, Cipher};
168    /// # let rk_256 = Key::rand_key_256()?;
169    /// # let cipher = Cipher::new(&rk_256);
170    /// let plaintext = ("Hello, World!").as_bytes();
171    /// let aad = ("Some data to be authenticated but not encrypted").as_bytes();
172    ///
173    /// let ciphertext_with_aad = cipher.encrypt_gcm(plaintext, Some(aad))?;
174    /// let ciphertext_no_aad = cipher.encrypt_gcm(plaintext, None)?;
175    /// # Ok(())
176    /// # }
177    /// ```
178    pub fn encrypt_gcm(&self, plaintext: &[u8], aad: Option<&[u8]>) -> Result<Vec<u8>> {
179        // generate random IV
180        let iv = random_iv()?;
181
182        // calculate AAD size
183        let aad_bytes = aad.unwrap_or(&[]);
184        let aad_len: u32 = aad_bytes
185            .len()
186            .try_into()
187            .expect("AAD size cannot exceed 2^32 bytes");
188
189        // initialise output vector
190        let mut out = Vec::with_capacity(12 + 4 + aad_bytes.len() + plaintext.len() + 16);
191
192        // build output: IV (12 bytes) || AAD length (4 bytes) || AAD || Ciphertext || Tag (16 bytes)
193        out.extend_from_slice(&iv);
194        out.extend_from_slice(&aad_len.to_be_bytes());
195        out.extend_from_slice(aad_bytes);
196
197        let mut ct = ctr_core(plaintext, &self.round_keys, &iv, 2)?;
198        let tag = compute_tag(&ct, &self.round_keys, &iv, aad_bytes)?;
199
200        out.append(&mut ct);
201        out.extend_from_slice(&tag);
202        Ok(out)
203    }

205    /// **Galois/counter mode** decryption.
206    ///
207    /// Assumes input follows the same format as [encryption](crate::Cipher::encrypt_gcm):
208    /// `IV (12 bytes) || AAD length (4 bytes) || AAD || Ciphertext || Tag (16 bytes)`
209    ///
210    /// Returns:
211    /// - `(plaintext, AAD)` if tag was authenticated and decryption was successful.
212    /// - [AuthFailed](crate::Error::AuthFailed) error if computed tag did not match input tag.
213    /// - [CounterOverflow](crate::Error::CounterOverflow) error if more than 2^32 blocks were provided.
214    /// - [InvalidCiphertext](crate::Error::InvalidCiphertext) error if ciphertext does not match expected format.
215    /// 
216    /// ## Examples
217    /// ```
218    /// # fn main() -> aesp::Result<()> {
219    /// # use aesp::{Key, Cipher};
220    /// # let rk_256 = Key::rand_key_256()?;
221    /// # let cipher = Cipher::new(&rk_256);
222    /// let plaintext = ("Hello, World!").as_bytes();
223    /// let aad = ("Some data to be authenticated but not encrypted").as_bytes();
224    ///
225    /// // Decryption with AAD
226    /// let ciphertext = cipher.encrypt_gcm(plaintext, Some(aad))?;
227    /// let (decrypted, returned_aad) = cipher.decrypt_gcm(&ciphertext)?;
228    ///
229    /// assert_eq!(decrypted, plaintext);
230    /// assert_eq!(returned_aad, Some(aad.to_vec()));
231    ///
232    /// // Decryption without AAD
233    /// let ciphertext = cipher.encrypt_gcm(plaintext, None)?;
234    /// let (_, returned_aad) = cipher.decrypt_gcm(&ciphertext)?;
235    /// assert!(returned_aad.is_none());
236    /// # Ok(())
237    /// # }
238    /// ```
239    pub fn decrypt_gcm(&self, ciphertext: &[u8]) -> Result<(Vec<u8>, Option<Vec<u8>>)> {
240        // minimum size is 32 bytes -> 12 (iv) + 4 (aad_len) + 16 (tag)
241        if ciphertext.len() < 32 {
242            return Err(Error::InvalidCiphertext {
243                len: ciphertext.len(),
244                context: "insufficient bytes for valid GCM",
245            });
246        }
247
248        // extract IV
249        let (iv_bytes, ciphertext) = ciphertext.split_at(12);
250        let mut iv = [0u8; 12];
251        iv.copy_from_slice(iv_bytes);
252
253        // extract AAD len and validate remaining size
254        let (aad_len, ciphertext) = ciphertext.split_at(4);
255        let aad_len = u32::from_be_bytes([aad_len[0], aad_len[1], aad_len[2], aad_len[3]]);
256        if ciphertext.len() < aad_len as usize + 16 {
257            return Err(Error::InvalidCiphertext {
258                len: ciphertext.len(),
259                context: "insufficient bytes given aad_len",
260            });
261        }
262
263        // extract aad and save in vector
264        let (aad, ciphertext) = ciphertext.split_at(aad_len as usize);
265        let aad = aad.to_vec();
266
267        // extract tag and format as [u8; 16]
268        let mut received_tag = [0u8; 16];
269        let (ct, tag_bytes) = ciphertext.split_at(ciphertext.len() - 16);
270        received_tag.copy_from_slice(tag_bytes);
271
272        // compute and compare tag
273        let computed_tag = compute_tag(ct, &self.round_keys, &iv, &aad)?;
274        if received_tag != computed_tag {
275            return Err(Error::AuthFailed);
276        }
277
278        // wrap AAD in option
279        let aad = if !aad.is_empty() { Some(aad) } else { None };
280
281        // run ctr starting at 2, as per NIST spec
282        let plaintext = ctr_core(ct, &self.round_keys, &iv, 2)?;
283        Ok((plaintext, aad))
284    }

286    /// AES key schedule. Returns a vector of 11, 13, or 15 round keys, corresponding with AES-128, AES-192,
287    /// and AES-256, respectively. The extra round key is the initial round key, which is not counted in most
288    /// documentation as it is simply the original key.
289    fn expand_key(key: &Key) -> Vec<[u8; 16]> {
290        let key = key.as_bytes();
291
292        // Variable names match FIPS-197, NIST specification: https://doi.org/10.6028/NIST.FIPS.197-upd1
293        // Nk   The number of 32-bit words comprising the key
294        // Nr   The number of rounds. 10, 12, and 14 for AES-128, AES-192, and AES-256, respectively
295        // w    The result of the key schedule, an array of words that form round keys
296        // Nw   The total number of words generated by the key schedule (including initial key)
297        let nk = key.len() / 4; // key size (in 4-byte words)
298        let nr = nk + 6; // number of rounds = num of words in key + 6
299        let nw = (nr + 1) * 4; // total number of words resulting from expansion
300
301        // initialise w, vector comprising 4-byte words of round_keys
302        let mut w: Vec<[u8; 4]> = vec![[0u8; 4]; nw];
303
304        // first nk words of w are filled with the initial key
305        for i in 0..key.len() {
306            w[i / 4][i % 4] = key[i];
307        }
308
309        // initialise temp variable
310        let mut temp = w[nk - 1];
311        for i in nk..nw {
312            if i % nk == 0 {
313                // calculate rot_word, sub_word, and rcon on temp
314                temp = [
315                    SBOX[temp[1] as usize] ^ RCON[i / nk],
316                    SBOX[temp[2] as usize],
317                    SBOX[temp[3] as usize],
318                    SBOX[temp[0] as usize],
319                ];
320            } else if nk == 8 && i % nk == 4 {
321                // additional substitution on temp for AES-256 only
322                temp = [
323                    SBOX[temp[0] as usize],
324                    SBOX[temp[1] as usize],
325                    SBOX[temp[2] as usize],
326                    SBOX[temp[3] as usize],
327                ];
328            }
329
330            // w[i] = temp ⊕ w[i − Nk]
331            for b in 0..4 {
332                w[i][b] = temp[b] ^ w[i - nk][b];
333            }
334            
335            temp = w[i]; // update temp
336        }
337
338        // convert words vector into indexable round_keys vector
339        let mut round_keys = vec![[0u8; 16]; nr + 1];
340        for round in 0..=nr {
341            let base = round * 4;
342            for col in 0..4 {
343                let word = w[base + col];
344                for row in 0..4 {
345                    round_keys[round][col * 4 + row] = word[row];
346                }
347            }
348        }
349
350        round_keys
351    }
