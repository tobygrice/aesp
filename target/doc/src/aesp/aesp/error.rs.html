1use thiserror::Error;
2use rand::rand_core;

4/// AES Result type.
5pub type Result<T> = std::result::Result<T, Error>;
6
7/// AES Error type. 
8#[derive(Debug, Error)]
9#[non_exhaustive]
10pub enum Error {
11    /// Computed authentication tag did not match input tag. Ciphertext and/or AAD has been modified since it was encrypted.
12    #[error("GCM authentication failed (invalid tag)")]
13    AuthFailed,
14
15    /// Attempted to encrypt or decrypt more than 2^32 16-byte blocks in GCM/CTR mode.
16    #[error("input size caused counter overflow (maximum input size for 32 bit counter is 16 * 2^32 bytes)")]
17    CounterOverflow,
18    
19    /// Attempted to instantiate an AES key with an input size that is not 128, 192, or 256 bits.
20    #[error("invalid key length: {len} bytes (expected 16, 24, or 32)")]
21    InvalidKeyLength { len: usize },
22
23    /// Provided ciphertext that did not match the expected format of the mode of operation.
24    #[error("invalid ciphertext length: {len} bytes ({context})")]
25    InvalidCiphertext { len: usize, context: &'static str },
26
27    /// Provided plaintext that did not match the expected format of the mode of operation.
28    #[error("invalid ECB input length: {len} bytes (must be a multiple of 16)")]
29    InvalidECBInput { len: usize },
30
31    /// OS RNG failed during random key generation.
32    #[error("OS RNG failed in random key generation")]
33    Rng(#[from] rand_core::OsError),
34}
